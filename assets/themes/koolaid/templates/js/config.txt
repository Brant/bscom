function siteVars(opt){
	switch (opt){
		case 'home':
			return '{% url home %}';
			break;
		case 'static':
			return '{{ settings.STATIC_URL }}';
			break;
		case 'img':
			return siteVars('theme') + 'img/';
			break; 
		case 'css':
			return siteVars('theme') + 'css/';
			break;
		default:
			return null;
			break;
	}
}
